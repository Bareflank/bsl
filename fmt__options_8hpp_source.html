574  constexpr void
575  fmt_options_impl_width(
576  cstr_type const f, bsl::uintmax &idx, bsl::uintmax const len) noexcept
577  {
578  constexpr bsl::uintmax max_num_digits{3U};
579 
580  for (bsl::uintmax i{}; (i < max_num_digits) && (idx < len); ++i) {
581  char_type const digit{f[idx]};
582 
583  if ((digit >= '0') && (digit <= '9')) {
584  m_width *= 10U;
585  m_width += static_cast<bsl::uintmax>(digit);
586  m_width -= static_cast<bsl::uintmax>('0');
587  ++idx;
588  }
589  }
